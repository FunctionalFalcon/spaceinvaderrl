# Pinned for Colab / Kaggle reproducibility.
# Tested against Python 3.10 (Colab default) and Python 3.11 (Kaggle default).
# Also works on Python 3.12 (current Colab default as of 2026).
#
# How Atari is wired:
#   gymnasium.make("ALE/SpaceInvaders-v5", ...) uses the Shimmy ALE env.
#   gymnasium 0.29.1 does NOT pull shimmy/ale-py as core deps — Atari is
#   an optional extra. We install shimmy + ale-py directly with versions
#   known to coexist on Python 3.12.
#
# Why these specific versions:
#   - shimmy 0.x pins ale-py~=0.8.x via the [atari] extra, and ale-py 0.8.x
#     has no cp312 wheels -> ResolutionImpossible on Python 3.12.
#   - shimmy 1.3.0 is a universal wheel; its [atari] extra STILL pins
#     ale-py~=0.8.1, so we install it WITHOUT the [atari] extra.
#   - ale-py 0.10.2 has cp312 manylinux wheels and is API-compatible
#     with what gymnasium.make("ALE/SpaceInvaders-v5", ...) needs
#     (frameskip, repeat_action_probability).

# Core gymnasium (no [atari] extra)
gymnasium==0.29.1

# ALE bridge — base shimmy, NOT shimmy[atari]. The [atari] extra on any
# shimmy version pins ale-py to ~=0.8.x, which has no cp312 wheel.
shimmy==1.3.0

# Atari backend (ALE-Python). 0.10.2 is the first with cp312 manylinux.
ale-py==0.10.2

# RL
stable-baselines3==2.3.0

# Deep learning.
# NOTE: We deliberately do NOT pin torch / torchvision here. Colab and Kaggle
# both preinstall a working CUDA build, and SB3 only needs torch >= 1.13.
# Pinning to a CUDA-suffixed build (e.g. torch==2.1.0+cu118) breaks pip
# resolution because those wheels live on pytorch.org, not PyPI.

# Numerics / vision.
numpy==1.26.4
opencv-python==4.9.0.80

# Logging
tensorboard==2.16.2

# Optional: for the record_random.py / play.py video capture path
imageio==2.34.0
imageio-ffmpeg==0.4.9
