# Pinned for Colab / Kaggle reproducibility.
# Tested on Python 3.12 (current Colab default as of 2026).
#
# Why these versions:
#   - gymnasium 1.x: required by shimmy 2.x. 0.29.1 + shimmy 1.3.0 + ale-py
#     0.10.2 hit two unfixable issues on Python 3.12: (1) shimmy's plugin
#     loader races against gymnasium's import and silently fails to register
#     the ALE envs; (2) shimmy 1.3.0 imports `ale_py.roms.utils`, which was
#     removed in ale-py 0.10.x. Upgrading the whole stack to the modern
#     line (gymnasium 1.3.0 / shimmy 2.0.1 / ale-py 0.10.2) resolves both.
#   - shimmy 2.0.1 is the current release; it auto-registers Atari envs
#     correctly on import, so no explicit register_gymnasium_envs() call
#     is needed in user code.
#   - ale-py 0.10.2 is the first with cp312 manylinux wheels and is
#     API-compatible with gymnasium.make("ALE/SpaceInvaders-v5", ...)
#     (frameskip, repeat_action_probability).

# Core gymnasium
gymnasium==1.3.0

# ALE bridge — base shimmy (the [atari] extra is unnecessary on 2.x)
shimmy==2.0.1

# Atari backend (ALE-Python). 0.10.2 has cp312 manylinux wheels.
ale-py==0.10.2

# RL
stable-baselines3==2.3.0

# Deep learning.
# NOTE: We deliberately do NOT pin torch / torchvision here. Colab and Kaggle
# both preinstall a working CUDA build, and SB3 only needs torch >= 1.13.
# Pinning to a CUDA-suffixed build (e.g. torch==2.1.0+cu118) breaks pip
# resolution because those wheels live on pytorch.org, not PyPI.

# Numerics / vision. Unpinned numpy — pip's resolver picks a version
# compatible with shimmy 2.x's numpy constraint and SB3's >=1.20.
numpy>=1.26,<2.0
opencv-python==4.9.0.80

# Logging
tensorboard==2.16.2

# Optional: for the record_random.py / play.py video capture path
imageio==2.34.0
imageio-ffmpeg==0.4.9
