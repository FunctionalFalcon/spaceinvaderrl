# Pinned for Colab / Kaggle reproducibility.
# Tested against Python 3.10 (Colab default) and 3.11 (Kaggle default).
#
# IMPORTANT: Do NOT use `gymnasium[atari]==0.29.1` here. The [atari] extra
# pulls in `shimmy[atari]` whose pinned ale-py range (~=0.8.x) combined
# with numpy 1.26 triggers a ResolutionImpossible conflict on Python 3.12.
# Install shimmy + ale-py explicitly instead, with versions known to
# coexist. Same behavior, no resolver fight.

# Core gymnasium (no [atari] extra)
gymnasium==0.29.1

# ALE bridge + ROM-bearing Atari backend (Python 3.12 compatible versions)
shimmy[atari]==0.2.1
ale-py==0.8.1

# RL
stable-baselines3==2.3.0

# Deep learning.
# NOTE: We deliberately do NOT pin torch / torchvision here. Colab and Kaggle
# both preinstall a working CUDA build, and SB3 only needs torch >= 1.13.
# Pinning to a CUDA-suffixed build (e.g. torch==2.1.0+cu118) breaks pip
# resolution because those wheels live on pytorch.org, not PyPI.
# If you need to install torch yourself (e.g. local Linux box), do it BEFORE
# pip-installing the rest:
#   pip install torch torchvision --index-url https://download.pytorch.org/whl/cu118
# Then leave these lines commented or remove them.

# Numerics / vision. numpy<2 is required for shimmy 0.2.1 + opencv-python 4.9
# on Python 3.12. numpy 2.x breaks shimmy's C extensions.
numpy==1.26.4
opencv-python==4.9.0.80

# Logging
tensorboard==2.16.2

# Optional: for the record_random.py / play.py video capture path
imageio==2.34.0
imageio-ffmpeg==0.4.9
